import osimport sysimport numpy as npimport pandas as pdimport matplotlib.pyplot as pltfrom scipy.signal import detrendfrom scipy import stats# 工作目录work_dir = r" /EMA_Results"# 全局绘图设置color1_line = '#000000'ld = 2.2ft = 16plt.rc('font', family='serif')def AR_func(a1, a2):    """    计算两个数组的相关性（以各自标准差归一化）。    """    std_a1 = np.std(a1)    std_a2 = np.std(a2)    if std_a1 == 0 or std_a2 == 0:        return 0    value = np.mean(((a1 - np.mean(a1)) / std_a1) * ((a2 - np.mean(a2)) / std_a2))    return valuedef CalAR1(var, length):    """    使用滑动窗口计算 AR1 序列。    """    if length >= len(var):        return []    TPresid2 = var    AR1_result = []    for i in range(len(TPresid2) - length):        a1 = TPresid2[i:i + length]        a2 = TPresid2[i + 1:i + length + 1]        AR1_result.append(AR_func(a1, a2))    return AR1_resultdef PlotOriDat(ax, time_data, value_data, x_label_name, y_label_name):    """    绘制原始数据，显示所有 x 轴刻度。    """    ax.plot(time_data, value_data, linestyle='-', linewidth=1.5, color=color1_line)    ymin = np.min(value_data)    ymax = np.max(value_data)    yM = ymax + (ymax - ymin) * 0.1    ym = ymin - (ymax - ymin) * 0.1    ax.set_ylim([ym, yM])    ax.set_xlim([time_data[0], time_data[-1]])    # 显示所有点作为刻度，旋转90度避免重叠    ax.set_xticks(time_data)    ax.set_xticklabels([f"{t:.2f}" for t in time_data], rotation=90, fontsize=8)    ax.set_xlabel(x_label_name)    ax.set_ylabel(y_label_name)def PlotAnaDat1(ax, time_data, value_data, x_label_name, y_label_name, ylashi):    """    绘制分析数据（散点图），显示所有 x 轴刻度。    """    ax.scatter(time_data, value_data, marker='o', linestyle='-', color=color1_line, s=50)    ymin = np.min(value_data)    ymax = np.max(value_data)    yM = ymax + (ymax - ymin) * 0.1    ym = ymin - (ymax - ymin) * 0.1    ax.set_ylim([ym, yM])    ax.set_xlim([time_data[0], time_data[-1]])    ax.set_xticks(time_data)    ax.set_xticklabels([f"{t:.2f}" for t in time_data], rotation=90, fontsize=8)    ax.set_xlabel(x_label_name)    ax.set_ylabel(y_label_name, labelpad=ylashi)# ---------- 主程序 ----------input_file_name = "Species_EMA_Analysis_Mode_Contribution_Entropy.csv"Curdir = work_dirprint(f"Reading and preparing data from: {os.path.join(Curdir, input_file_name)}")try:    file_path = os.path.join(Curdir, input_file_name)    df = pd.read_csv(file_path)except FileNotFoundError:    print(f"Error: File not found at {file_path}.")    sys.exit(1)except Exception as e:    print(f"Error reading {file_path}: {e}.")    sys.exit(1)# 按 Eh 值排序df.sort_values(by='EhValue_Label', inplace=True)time_eh = np.array(df['EhValue_Label'].astype(float))val_entropy = np.array(df['Mode_Contribution_Entropy_bits'].astype(float))val_entropy_detrended = detrend(val_entropy)# 存放原始序列与去趋势序列Val = np.zeros([2, len(val_entropy)])Val[0, :] = val_entropyVal[1, :] = val_entropy_detrendedbase_name = os.path.splitext(input_file_name)[0]# 遍历不同窗口大小window_range = range(3, 21)print(f"\nStarting batch processing for window sizes from {window_range.start} to {window_range.stop - 1}...")for windows in window_range:    print(f"--- Processing for window size = {windows} ---")    if windows >= len(val_entropy):        print(f"Skipping window size {windows} because it is too large.")        continue    fig, ax = plt.subplots(1, 2, figsize=(15, 6))    fig.suptitle(f"{base_name} (Window={windows})", fontsize=ft)    # 左图：原始熵序列    PlotOriDat(ax[0], time_eh, Val[0, :], "Eh", "Mode_Contribution_Entropy_bits")    # 计算滑动窗口 AR1（基于去趋势序列）    wt = time_eh[windows // 2:len(time_eh) - (windows - windows // 2)]    AR1Val = CalAR1(Val[1, :], windows)    if len(wt) != len(AR1Val):        min_len = min(len(wt), len(AR1Val))        wt = wt[:min_len]        AR1Val = AR1Val[:min_len]    # 右图：AR1 值    shifty = 0    PlotAnaDat1(ax[1], wt, AR1Val, "Eh", "AR1", shifty)    plt.tight_layout(rect=[0, 0.03, 1, 0.95])    output_filename = f"{base_name}_win{windows}.pdf"    output_filepath = os.path.join(Curdir, output_filename)    plt.savefig(output_filepath, dpi=300, bbox_inches='tight')    print(f"Plot saved to: {output_filepath}")    plt.close(fig)print("\nProcessing complete. All plots have been generated.")
