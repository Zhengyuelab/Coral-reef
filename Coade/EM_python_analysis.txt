{raw_data_for_processing.shape[1]}")# ========================# 2. 辅助函数定义# ========================def extract_timepoint(col_name):    """    尝试从列名中提取 Eh 值或时间点标识符。    纯数字 -> 原样返回；    'Eh-10_rep1' -> 'Eh-10';    'T1_rep1' -> 'T1';    其他 -> 返回第一个部分。    """    try:        float(col_name)        return col_name    except ValueError:        parts = col_name.split('_')        if len(parts) > 0:            potential_label = parts[0]            try:                float(potential_label.replace('Eh', '').replace('T', ''))                return potential_label            except ValueError:                return potential_label        else:            return col_name    except Exception:        return col_namedef visualize_and_save(fig, filename_base, level, 开始处理层级: {level_name} ---")    print(f"输入数据维度 ({level_name} x 样本): {data_counts.shape}")    if data_counts.empty or data_counts.shape[0] < 2 or data_counts.shape[1] < 2:        print(f"警告: 层级 {level_name} 的数据不足，跳过此层级。")        return None    # 预处理：伪计数 -> 相对丰度 -> CLR    data_counts_pseudo = data_counts + pseudo_count    sample_pseudo_totals = data_counts_pseudo.sum(axis=0)    sample_pseudo_totals[sample_pseudo_totals <= epsilon] = epsilon    data_rel = data_counts_pseudo.div(sample_pseudo_totals, axis=1)    data_rel_transposed = data_rel.T    if np.any(data_rel_transposed <= 0):        print(f"警告: CLR前相对丰度含零/负值。")    try:        clr_transformed_values = clr(data_rel_transposed.values)        if not np.all(np.isfinite(clr_transformed_values)):            nan_count = np.sum(np.isnan(clr_transformed_values))            inf_count = np.sum(np.isinf(clr_transformed_values))            print(f"错误：CLR 变换结果包含 {nan_count} 个 NaN 和 {inf_count} 个 Inf。"                  f"无法继续处理层级 {level_name}。")            return None    except MemoryError:        print(f"错误：CLR变换时内存不足。数据维度 ({data_rel_transposed.shape}) 可能过大。")        return None    except Exception as e:        print(f"错误: CLR变换失败: {e}")        return None    data_clr = pd.DataFrame(clr_transformed_values.T,                            index=data_rel.index,                            columns=data_rel_transposed.index)    # 按提取的 Eh 标签聚合（求均值），并按外部排序重排    data_clr.columns = data_clr.columns.map(extract_timepoint)    merged_clr_data = data_clr.groupby(axis=1, level=0).mean()    merged_clr_data = merged_clr_data.reindex(columns=eh_order).dropna(axis=1, how='all')    if np.sum(np.isnan(X))        inf_count = np.sum(np.isinf(X))        print(f"错误：合并后的CLR数据(X)包含 {nan_count} 个 NaN 和 {inf_count} 个 Inf。")        return None    N_taxa = X.shape[0]    N_eh_points = X.shape[1]    if N_taxa < 2 or N_eh_points < 2:        print(f"错误: 最终矩阵维度不足({N_taxa}, {N_eh_points})。")        return None    # 对相对丰度也做同样的聚合，用于后续绘图    data_rel.columns = print(f"合并后相对丰度维度: {merged_rel_data.shape}")    # EMA 计算    print(f"开始EMA计算 ({level_name})...")    means_per_taxa = X.mean(axis=1, keepdims=True)    dX = X - means_per_taxa    if not len(zero_std_indices) > 0:        print(f"警告：发现 {len(zero_std_indices)} 个类群/物种在 CLR 空间中的标准差接近于零。")        print("  这些类群对动态贡献可能很小，已将其标准差替换为 epsilon。")    stds_per_taxa[stds_per_taxa < epsilon] = epsilon    dS = dX / stds_per_taxa[:, np.newaxis]    if not np.all(np.isfinite(dS)):        nan_count = np.sum(np.isnan(dS))        inf_count = np.sum(np.isinf(dS))        print(f"错误：标准化涨落矩阵(dS)包含 {nan_count} 个 NaN 和 {inf_count} 个 Inf。")        return None    Co = np.sum(dS ** 2)    if Co <= epsilon:        print(f"错误: Co ({Co}) 过小。")        return None    A = dS / np.sqrt(Co)    if not np.all(np.isfinite(A)):        nan_count = np.sum(np.isnan(A))        inf_count = np.sum(np.isinf(A))        print(f"错误：最终的系综矩阵 A 包含 {nan_count} 个 NaN 和 {inf_count} 个 Inf。")        return None    try:        U, sigma, Vt = svd(A, full_matrices=False)        print("SVD分解完成。")    except np.linalg.LinAlgError as e:        print(f"错误: SVD分解时发生线性代数错误: {e}")        return None    except ValueError as e:        print(f"错误: SVD分解时发生值错误: {e}")        return None    except Exception as e:        print(f"错误: SVD 分解时发生未知错误: {e}")        return None    species_patterns = U    time_evolutions = Vt.T    contributions = sigma ** 2    contribution_sum = contributions.sum()    print(f"所有模式贡献率 (sigma^2) 总和: {contribution_sum:.6f}")    if not np.isclose(contribution_sum, 1.0, atol=1e-3):        print(f"警告：贡献率总和 ({contribution_sum:.6f}) 与 1.0 相差较多 (> 1e-3)。"              "分析将继续，不进行手动缩放。")    print("-" * 20)    print(f"层级 {level_name} 计算模式总数: {len(contributions)}")    for i in range(min(n_modes_to_analyze, len(contributions))):        print(f"Mode {i + 1} Contribution: {contributions[i] * 100:.2f}%")    print("-" * 20) f"{non_numeric_labels}")    if not eh_order:        raise ValueError("未能从OTU表列名中提取到任何可用于分析的有效Eh值。")    print(f"\n检测并排序的Eh值: {eh_order}")except Exception as e:    print(f"错误：处理Eh值排序时出错: {e}")    exit()# 主分析循环for level in taxonomic_levels_to_analyze:    current_output_prefix = f"{level}_{output_prefix_base}"    print(f"\n{'='*15} 分析层级: {level} {'='*15}")    # 数据准备    if level ==       fill_value = f"Unassigned {level}"        merged_for_agg[level] = merged_for_agg[level].fillna(fill_value)        replace_map = ['', 'na', 'NA', 'N/A', 'none', 'unknown', '-', 'Unassigned', {current_data_counts.shape[0]}")            if current_data_counts.shape[0] < 2:                print(f"错误: 剩余类群不足2。跳过 {level}")                report_summary[level] = "Skipped (Less than 2 taxa)"                continue        except Exception as e:            print(f"错误: 聚合 {level} 失败: {e}")            report_summary[level] = "Skipped (Aggregation Error)"            continue    # 运行
